name1 = name1.lower()

word_love = 0

word_true += name1.count("r") + name2.count("r")
word_true += name1.count("u") + name2.count("u")
word_true += name1.count("e") + name2.count("e")

word_love += name1.count("o") + name2.count("o")
word_love += name1.count("v") + name2.count("v")
word_love += name1.count("e") + name2.count("e")

word_true = str(word_true)
word_love = str(word_love)
score = word_true + word_love
score = int(score)
if score < 10 or 90 < score:
    print(f"Your score is {score}, you go together like coke and mentos.")

    print(f"Your score is {score}, you are alright together.")

    print(f"Your score is {score}.")
